 # to create user Raymond and John 
 sudo user add -m -d /home/raymond -g users -s /bin/bash raymond
 sudo user add -m -d /home/john -g users -s /bin/bash john
 sudo passwd john
2.
 # generate key for Raymond 
 ssh-keygen -t ed25519 -C "raymond@localhost"
 sudo mkdir -p /home/Raymond/.ssh

 # here we need to paste generated key for raymond
 sudo nano /home/Raymond/.ssh/authorized_keys

 # to set configurations
 sudo nano /etc/ssh/sshd_config

 Match User raymond
    PasswordAuthentication no
    PubkeyAuthentication yes

 Match User john
    PasswordAuthentication yes
    PubkeyAuthentication no

 # to reset the ssh
 sudo systemctl restart ssh
 
 # to connect
 ssh raymond@localhost
 ssh john@localhost
